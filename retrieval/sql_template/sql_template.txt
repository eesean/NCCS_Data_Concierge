HEADER: How many deaths occurred in 2020?
TIP: aggregation, filtering
SQL:
SELECT COUNT(*) AS death_count FROM death WHERE YEAR(death_date) = 2020;

HEADER: List the top 5 races based on number of person records. Order them in descending order of number of records.
TIP: ordering, limit, aggregation
SQL:
SELECT race_source_value, COUNT(*) AS race_count
FROM person GROUP BY race_source_value ORDER BY race_count DESC LIMIT 5;

TIP: join, aggregation
SQL:
SELECT gender_source_value, COUNT(*) AS gender_count FROM person LEFT JOIN condition_occurrence USING (person_id) WHERE ICD10 = 'C18.7' AND YEAR(condition_start_date) = 2021 GROUP BY gender_source_value;

HEADER: Consolidate number of measurement, procedure and drug exposure events respectively.
TIP: union
SQL:
SELECT 'measurement' AS event_type, COUNT(*) AS event_count FROM measurement_mutation
UNION ALL
SELECT 'procedure' AS event_type, COUNT(*) AS event_count FROM procedure_occurrence

SELECT 'drug_exposure' AS event_type, COUNT(*) AS event_count FROM drug_exposure_cancerdrugs;

HEADER: Create a table reporting the number of procedure and drug exposure events respectively in each institution.
TIP: join, aggregation, nested queries
SQL:
SELECT t1.institution,
t1.drug_exposure_events,
COALESCE(t2.procedure_occurrence_events, 0) AS procedure_occurrence_events
FROM (
    SELECT institution,
    COUNT(*) AS drug_exposure_events
    FROM drug_exposure_cancerdrugs GROUP BY institution
) t1
LEFT JOIN (
    SELECT institution,
    COUNT(*) AS procedure_occurrence_events
    FROM procedure_occurrence GROUP BY institution
) t2
USING (institution);

HEADER: How many procedures in 2020 lasted for longer than a week?
TIP: datetime processing
SQL:
SELECT COUNT(*) AS count FROM procedure_occurrence
WHERE DATEDIFF('day',
    STRPTIME(procedure_datetime, '%d/%m/%Y %H:%M'),
    STRPTIME(procedure_end_datetime, '%d/%m/%Y %H:%M')
) > 3
AND YEAR(STRPTIME(procedure_datetime, '%d/%m/%Y %H:%M')) = 2020
AND YEAR(STRPTIME(procedure_end_datetime, '%d/%m/%Y %H:%M')) = 2020;

HEADER: What percentage of drug exposure events recorded in 2019 involve inpatients?
TIP: arithmetic, sum, case when, datetime processing
SQL:
SELECT ROUND(inpatient_count / NULLIF(total_count, 0) * 100, 2) AS inpatient_percentage
FROM (
    SELECT SUM(CASE WHEN LOWER(case_type) = 'inpatient' THEN 1 ELSE 0 END) AS inpatient_count,
    COUNT(*) AS total_count
    FROM drug_exposure_cancerdrugs
    WHERE YEAR(drug_exposure_start_date) = 2019 AND YEAR(drug_exposure_end_date) = 2019
);
